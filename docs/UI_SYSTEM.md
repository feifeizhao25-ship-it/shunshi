```dart
enum MessageType {
  text,           // 纯文字
  suggestionCard, // 建议卡片
  acupointCard,   // 穴位卡片
  dietCard,       // 食疗卡片
  solarTermCard,  // 节气卡片
  followUp,       // 跟进提醒
  voice,          // 语音消息
}

class ChatMessage {
  final String id;
  final MessageType type;
  final String content;
  final DateTime timestamp;
  final bool isUser;
  final CardData? card;
}
```
